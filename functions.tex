% Skill level
\colorlet{fullcolor}{red}
\colorlet{emptcolor}{gray}
\newcount\skillcnt
\newcommand\skill[1]{
    \skillcnt=0
    \loop\ifnum\skillcnt<#1
        \advance\skillcnt1
        \textcolor{fullcolor}{$\bullet$}
    \repeat
    \loop\ifnum\skillcnt<5
        \advance\skillcnt1
        \textcolor{emptcolor}{$\bullet$}
    \repeat
}
\newcommand\skillname[2]{
    \hbox{
        \makebox[2cm][l]{#1}
        \skill{#2}
    }
}
